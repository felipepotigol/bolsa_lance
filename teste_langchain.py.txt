from langchain_core.prompts import ChatPromptTemplate

prompt = ChatPromptTemplate.from_messages([
    (
        "system",
        """
        Você é um pesquisador especialista em Aprendizagem Federativa.

        Explique conceitos relacionados a:
        - Federated Learning
        - Kubernetes
        - Machine Learning Distribuído
        - Edge Computing
        """
    ),
    ("human", "{pergunta}")
])

mensagens = prompt.invoke(
    {
        "pergunta": "O que é Aprendizagem Federativa?"
    }
)

print(mensagens)
